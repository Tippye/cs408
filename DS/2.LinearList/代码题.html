ListNode *diffTwoLists(ListNode *list1, ListNode *list2) {
    ListNode *p1 = list1->next;
    ListNode *p2 = list2->next;
    ListNode *pre = list1;


    while (p1 && p2) {
        if (p1->val == p2->val) {
            pre->next = p1->next;
            p1 = p1->next;
            p2 = p2->next;
        } else if (p1->val > p2->val) {
            ListNode *temp = p2;
            p2 = p2->next;
            pre->next = temp;
            temp->next = p1;
            pre = pre->next;
        } else {
            pre = p1;
            p1 = p1->next;
        }
    }
    if (!p1)
        pre->next = p2;


    return list1;
}
